#From Qiime2 to phyoseq
physeq <- qza_to_phyloseq(
  features="feature-table.qza",
  tree="rooted-tree.qza",
  taxonomy="taxonomy.qza",
  metadata = "New_sample_metadata.tsv")
####################################subset the samples#############################################

# Let's remove the control samples 
physeq = subset_samples(physeq, type != "0")
physeq = subset_samples(physeq, donor.id != "0")
physeq = subset_samples(physeq, type != "control")
#To relative 
physeq <- normalize(physeq, method = "relative")
## Normalization using relative method
#let's profile on the genus level our taxa
#for easy coding, sometimes I make than one variable for the same object 
ps = physeq
#Now, you have #Species #Genus
ps.rel = transform_sample_counts(ps, function(x) x/sum(x)*100)
# agglomerate taxa
glom <- tax_glom(ps.rel, taxrank = 'Genus', NArm = FALSE)
ps.melt <- psmelt(glom)
ps.melt$Genus <- as.character(ps.melt$Genus)
#grouped by CST
ps.melt <- ps.melt %>%
  group_by(CST, Genus) %>%
  mutate(median=median(Abundance))
keep <- unique(ps.melt$Genus[ps.melt$median > 1])
ps.melt$Genus[!(ps.melt$Genus %in% keep)] <- "Others"
ps.melt_sum <- ps.melt %>%
  group_by(Sample,CST,Genus) %>%
  summarise(Abundance=sum(Abundance))
## `summarise()` has grouped output by 'Sample', 'CST'. You can override using the
## `.groups` argument.
ggplot(ps.melt_sum, aes(x = Sample, y = Abundance, fill = Genus)) + 
  geom_bar(stat = "identity", aes(fill=Genus)) + 
  labs(x="", y="%") +
  facet_wrap(~CST, scales= "free_x", nrow=1) +
  theme_classic() + 
  scale_x_discrete(labels = NULL, breaks = NULL) + labs(x = "")+
  theme( 
        axis.text.x.bottom = element_text(angle = -90),
        axis.title.x=element_blank(),
        axis.text.x=element_blank(),
        axis.ticks.x=element_blank())

###########################################################################
#grouped by sex
ps.melt <- ps.melt %>%
  group_by(sex, Genus) %>%
  mutate(median=median(Abundance))
keep <- unique(ps.melt$Genus[ps.melt$median > 1])
ps.melt$Genus[!(ps.melt$Genus %in% keep)] <- "Others"
ps.melt_sum <- ps.melt %>%
  group_by(Sample,sex,Genus) %>%
  summarise(Abundance=sum(Abundance))
